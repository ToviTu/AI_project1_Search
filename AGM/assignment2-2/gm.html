# Define the function and the gradient
def evaluateFunc(x):
    return (1/n) * np.sum(np.log(1 + np.exp(-b * (A.T @ x))))

def evaluateGrad(x):
    return (1/n) * A @ (-b * np.exp(-b * (A.T @ x)) / (1 + np.exp(-b * (A.T @ x))))

for iter in range(maxIter):
    # gradient at w
    grad = evaluateGrad(x)

    # update
    xNext = x - stepSize * grad

    # evaluate the objective
    funcNext = evaluateFunc(xNext)

    # store the objective and the classification error
    objVals[iter] = funcNext
    
    # print('[%d/%d] [step: %.1e] [objective: %.4e]' % (iter, maxIter, stepSize, objVals[iter]))
    
    # check stopping criterion
    if np.linalg.norm(grad) < tol:
        break
    
    # update x
    x = xNext

def evaluatehSG(x):
    return 0.02 * np.sign(x)

def evaluategSG(x):
    return evaluateGrad(x)

def evaluatefSG(x):
    return evaluatehSG(x) + evaluategSG(x)

def evaluateFunch(x):
    return 0.02 * np.linalg.norm(x, 1)

def evaluateFuncg(x):
    return evaluateFunc(x)

def evaluateFuncf(x):
    return evaluateFunch(x) + evaluateFuncg(x)

def evaluateFuncSmoothh(x, mu=0.01):
    huber_loss = np.where(
        np.abs(x) <= mu,
        x^2 / (2 * mu),
        np.abs(x) - mu / 2
    )
    
    return 0.02 * np.sum(huber_loss)

def evaluateSmoothhGrad(x, mu=0.01):
    return 0.02 * np.where(
        np.abs(x) <= mu,
        x / mu,
        np.sign(x)
    )

def evaluateSmoothfGrad(x, mu=0.01):
    return evaluateGrad(x) + evaluateSmoothhGrad(x, mu=mu)
